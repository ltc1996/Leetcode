0965. 单值二叉树

- 简单

如果二叉树每个节点都具有相同的值，那么该二叉树就是*单值二叉树*。
只有给定的树是单值二叉树时，才返回 `true`；否则返回 `false`。


- 示例1：
![](https://assets.leetcode-cn.com/aliyun-lc-upload/uploads/2018/12/29/screen-shot-2018-12-25-at-50104-pm.png)

```
输入：[1,1,1,1,1,null,1]
输出：true
```

- 示例2：
![](https://assets.leetcode-cn.com/aliyun-lc-upload/uploads/2018/12/29/screen-shot-2018-12-25-at-50050-pm.png)
```
输入：[2,2,2,5,2]
输出：false
```

**提示**:
1. 定树的节点数范围是 `[1, 100]`。
2. 每个节点的值都是整数，范围为 `[0, 99]` 。


## 我的解答：
```python
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    # 用时20ms
    def isUnivalTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        # 所有非空节点都要判断val
        if root.left and not root.right:
            if root.val != root.left.val:
                return False
            return self.isUnivalTree(root.left)
        if not root.left and root.right:
            if root.val != root.right.val:
                return False
            return self.isUnivalTree(root.right)
        if root.left and root.right:
            if root.val != root.left.val or root.val != root.right.val:
                return False
            return self.isUnivalTree(root.left) and self.isUnivalTree(root.right)
        else:
            return True
```

## 最快解答：
```python
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def isUnivalTree(self, root):
        if not root:
            return True
        if not root.left and not root.right:
            return True
        else:
            if root.left and not root.left.val == root.val:
                return False
            if root.right and not root.right.val == root.val:
                return False
        return self.isUnivalTree(root.left) and self.isUnivalTree(root.right)
```

```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    // 用时0ms
    bool isUnivalTree(TreeNode* root) {
        if (root == NULL)
            return 1;
        
        return helper(root,root->val);
    }
    int helper(TreeNode *root,int val){
        if (root == NULL)
            return 1;
        if (root->val != val)
            return 0;
        
        int left = helper(root->left,val);
        int right = helper(root->right,val);
        return left &&right;
    }
};
```
