0203. 移除链表元素

- 简单

删除链表中等于给定值 ***val*** 的所有节点。

- 示例：
```
输入: 1->2->6->3->4->5->6, val = 6
输出: 1->2->3->4->5
```

## 我的解答：
```python
# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution(object):
    def removeElements(self, head, val):
        # 用时88ms
        """
        :type head: ListNode
        :type val: int
        :rtype: ListNode
        """
        if not head:
            return head
        head.next = self.removeElements(head.next, val)
        return head.next if head.val == val else head
```

## 最快解答：
```python
class Solution(object):
    # 用时44ms
    def removeElements(self, head, val):
        top = ListNode(0)       # head之前new一个节点
        top.next = head
        cur = head              # 用cur遍历
        last = top
        while cur:
            if cur.val == val:
                cur = cur.next
                last.next = cur
            else:
                last = cur
                cur = cur.next
        return top.next
```

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeElements(ListNode* head, int val) {
        ListNode *node = new ListNode(0);
        node->next=head;
        ListNode *tmp=node;
        while (tmp!=nullptr && tmp->next!=nullptr) {
            if (tmp->next->val == val)
                tmp->next=tmp->next->next;
            else
                tmp=tmp->next;
        }
        return node->next;
    }
};
```
