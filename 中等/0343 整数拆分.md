0343. 整数拆分

- 中等

给定一个正整数 *n*，将其拆分为**至少**两个正整数的和，并使这些整数的乘积最大化。 返回你可以获得的最大乘积。

- 示例1：
```
输入: 2
输出: 1
解释: 2 = 1 + 1, 1 × 1 = 1。
```

- 示例2：
```
输入: 10
输出: 36
解释: 10 = 3 + 3 + 4, 3 × 3 × 4 = 36。
```

**说明**: 你可以假设 n 不小于 2 且不大于 58。


## 最快解答：
```python
class Solution(object):
    # 用时16ms
    def integerBreak(self, n):
    # 将 n 拆成尽可能多的3，剩下的：
    # 若是2， 则×2
    # 若是1， 则×4
        if n <= 3:
            return n - 1
        result = 1
        while n > 4:
            n -= 3
            result *= 3

        return n * result
```

```python
# 动态规划解法
class Solution(object):
    def integerBreak(self, n):
        """
        :type n: int
        :rtype: int
        """
        # res[i]: i 表示的最大乘积
        # 前几个数res[i] < i, 故取max
        res = [0] * (1 + n)
        for i in range(2, n + 1):
            for j in range(2, i + 1):
                res[i] = max(res[i], max(j, res[j]) * max(i - j, res[i - j]))
        print res
        return res[-1]
```

```python
class Solution(object):
    def integerBreak(self, n):
    # 动态规划dp[i] = max(dp[i-j] * j, dp[i] , (i-j)) j<i-1
    # 注意：拆分为1显然不是最小
    # 从2开始
        dp = [0] * (n + 1)
        dp[1], dp[2] = 1, 1
        for i in range(3, n + 1):
            for j in range(1, i):
                dp[i] = max(dp[i], max(dp[i - j], i - j) * j)
        return dp[-1]
```

```cpp
class Solution {
public:
    int integerBreak(int n) {
        vector<int> dp(n+1, 0);
        dp[1] = 1, dp[2] = 1;
        for(int i = 3; i <= n; i++){
            for(int j = 1; j < i; j++){
                dp[i] = max(dp[i], j * max(i - j, dp[i - j]));
            }
        }
        return dp[n];
    }
};
```
